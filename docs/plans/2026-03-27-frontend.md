**Goal:** Build a Next.js frontend with 3D interactive elements, glassmorphism UI, animated transitions — covering auth, listings dashboard with tilt cards, and a listing detail page with 3D photo orbit viewer and animated pipeline status.

**Architecture:** Next.js 15 App Router with TypeScript and Tailwind CSS. React Three Fiber powers 3D elements (floating house hero, photo orbit viewer, pipeline visualizer). **Tech Stack:** Next.js 15, React 19, TypeScript, Tailwind CSS 4, @react-three/fiber, @react-three/drei, three, framer-motion, Vitest, React Testing Library

## Key Patterns

### Design tokens (Tailwind)
All colors/fonts from the design system are configured in `tailwind.config.ts` or `globals.css`:
```css
:root {
  --color-primary: #2563EB;
  --color-secondary: #3B82F6;
  --color-cta: #F97316;
  --color-bg: #F8FAFC;
  --color-text: #1E293B;
}
```

### Glassmorphism pattern
```tsx
<div className="bg-white/70 backdrop-blur-xl border border-white/20 shadow-xl rounded-xl">
```

### 3D scene pattern
Every 3D element is wrapped in `SceneWrapper` which handles Canvas + Suspense + fallback:
```tsx
<SceneWrapper fallback={<div className="animate-pulse bg-gray-200 rounded-xl h-64" />}>
  <FloatingHouse />
</SceneWrapper>
```

### Framer Motion tilt card
```tsx
<motion.div
  style={{ rotateX, rotateY, transformStyle: "preserve-3d" }}
  whileHover={{ scale: 1.02 }}
  transition={{ type: "spring", stiffness: 300, damping: 20 }}
>
```

### Test strategy
3D components (Three.js) are not tested in jsdom — they're wrapped behind `SceneWrapper` which renders a fallback in test env. Tests focus on data rendering, interactions, and API integration.

```typescript
export class ApiError extends Error {
  constructor(public status: number, public detail: string) {
    super(detail);
    this.name = 'ApiError';
  }
}

export class ApiClient {
  private baseUrl: string;
  private token: string | null = null;

  constructor(baseUrl: string) {
    this.baseUrl = baseUrl;
  }

  setToken(token: string): void { this.token = token; }
  clearToken(): void { this.token = null; }
  getToken(): string | null { return this.token; }

  private headers(): Record<string, string> {
    const h: Record<string, string> = { 'Content-Type': 'application/json' };
    if (this.token) h['Authorization'] = `Bearer ${this.token}`;
    return h;
  }

  private async request<T>(path: string, options: RequestInit = {}): Promise<T> {
    const resp = await fetch(`${this.baseUrl}${path}`, {
      ...options,
      headers: { ...this.headers(), ...(options.headers as Record<string, string> || {}) },
    });
    if (!resp.ok) {
      const body = await resp.json().catch(() => ({ detail: 'Request failed' }));
      throw new ApiError(resp.status, body.detail || 'Request failed');
    }
    return resp.json();
  }

  get<T>(path: string): Promise<T> {
    return this.request<T>(path, { method: 'GET' });
  }

  post<T>(path: string, body?: unknown): Promise<T> {
    return this.request<T>(path, { method: 'POST', body: body ? JSON.stringify(body) : undefined });
  }

  patch<T>(path: string, body: unknown): Promise<T> {
    return this.request<T>(path, { method: 'PATCH', body: JSON.stringify(body) });
  }
}

const API_URL = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:8000';
export const apiClient = new ApiClient(API_URL);
```

```tsx
'use client';

import { motion, useMotionValue, useTransform } from 'framer-motion';
import { useRef, type ReactNode } from 'react';

interface GlassCardProps {
  children: ReactNode;
  className?: string;
  tilt?: boolean;
  onClick?: () => void;
}

export function GlassCard({ children, className = '', tilt = false, onClick }: GlassCardProps) {
  const ref = useRef<HTMLDivElement>(null);
  const x = useMotionValue(0);
  const y = useMotionValue(0);
  const rotateX = useTransform(y, [-0.5, 0.5], [8, -8]);
  const rotateY = useTransform(x, [-0.5, 0.5], [-8, 8]);

  const handleMouseMove = (e: React.MouseEvent) => {
    if (!tilt || !ref.current) return;
    const rect = ref.current.getBoundingClientRect();
    x.set((e.clientX - rect.left) / rect.width - 0.5);
    y.set((e.clientY - rect.top) / rect.height - 0.5);
  };

  const handleMouseLeave = () => {
    x.set(0);
    y.set(0);
  };

  return (
    <motion.div
      ref={ref}
      onMouseMove={handleMouseMove}
      onMouseLeave={handleMouseLeave}
      onClick={onClick}
      style={tilt ? { rotateX, rotateY, transformStyle: 'preserve-3d' } : undefined}
      whileHover={{ scale: tilt ? 1.02 : 1, transition: { type: 'spring', stiffness: 300, damping: 20 } }}
      className={`bg-white/70 backdrop-blur-xl border border-white/20 shadow-xl rounded-xl ${onClick ? 'cursor-pointer' : ''} ${className}`}
    >
      {children}
    </motion.div>
  );
}
```

```tsx
import { useRef } from 'react';
import { useFrame } from '@react-three/fiber';
import { Float, MeshDistortMaterial } from '@react-three/drei';
import * as THREE from 'three';

export function FloatingHouse() {
  const groupRef = useRef<THREE.Group>(null);

  useFrame((state) => {
    if (groupRef.current) {
      groupRef.current.rotation.y = Math.sin(state.clock.elapsedTime * 0.3) * 0.15;
      groupRef.current.position.y = Math.sin(state.clock.elapsedTime * 0.5) * 0.1;
    }
  });

  return (
    <Float speed={2} rotationIntensity={0.3} floatIntensity={0.5}>
      <group ref={groupRef} scale={1.2}>
        {/* Base / foundation */}
        <mesh position={[0, -0.6, 0]}>
          <boxGeometry args={[2.4, 0.15, 1.8]} />
          <meshStandardMaterial color="#94A3B8" metalness={0.3} roughness={0.7} />
        </mesh>

        {/* Main body */}
        <mesh position={[0, 0.1, 0]}>
          <boxGeometry args={[2, 1.2, 1.5]} />
          <meshStandardMaterial color="#F8FAFC" metalness={0.1} roughness={0.8} />
        </mesh>

        {/* Roof */}
        <mesh position={[0, 1.05, 0]} rotation={[0, 0, 0]}>
          <coneGeometry args={[1.6, 0.8, 4]} />
          <meshStandardMaterial color="#2563EB" metalness={0.4} roughness={0.5} />
        </mesh>

        {/* Door */}
        <mesh position={[0, -0.1, 0.76]}>
          <boxGeometry args={[0.4, 0.6, 0.02]} />
          <meshStandardMaterial color="#F97316" metalness={0.2} roughness={0.6} />
        </mesh>

        {/* Windows */}
```

```tsx
'use client';

import { createContext, useContext, useState, useEffect, ReactNode, useCallback } from 'react';
import { apiClient } from '@/lib/api-client';
import type { UserResponse, TokenResponse } from '@/lib/types';

interface AuthState {
  user: UserResponse | null;
  token: string | null;
  isLoading: boolean;
  login: (email: string, password: string) => Promise<void>;
  register: (email: string, password: string, name: string, companyName: string) => Promise<void>;
  logout: () => void;
}

const AuthContext = createContext<AuthState | null>(null);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<UserResponse | null>(null);
  const [token, setToken] = useState<string | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const fetchUser = useCallback(async (t: string) => {
    try {
      apiClient.setToken(t);
      const me = await apiClient.get<UserResponse>('/auth/me');
      setUser(me);
    } catch {
      localStorage.removeItem('token');
      apiClient.clearToken();
      setToken(null);
      setUser(null);
    }
  }, []);

  useEffect(() => {
    const saved = localStorage.getItem('token');
    if (saved) {
      setToken(saved);
      fetchUser(saved).finally(() => setIsLoading(false));
    } else {
      setIsLoading(false);
    }
  }, [fetchUser]);

  const login = async (email: string, password: string) => {
    const resp = await apiClient.post<TokenResponse>('/auth/login', { email, password });
    localStorage.setItem('token', resp.access_token);
    setToken(resp.access_token);
    await fetchUser(resp.access_token);
  };

  const register = async (email: string, password: string, name: string, companyName: string) => {
    const resp = await apiClient.post<TokenResponse>('/auth/register', {
      email, password, name, company_name: companyName,
    });
    localStorage.setItem('token', resp.access_token);
    setToken(resp.access_token);
    await fetchUser(resp.access_token);
  };

  const logout = () => {
    localStorage.removeItem('token');
    apiClient.clearToken();
    setToken(null);
    setUser(null);
  };

  return (
    <AuthContext.Provider value={{ user, token, isLoading, login, register, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

export function useAuth() {
  const ctx = useContext(AuthContext);
  if (!ctx) throw new Error('useAuth must be inside AuthProvider');
  return ctx;
}
```

```tsx
'use client';

import { useMemo } from 'react';
import { Text } from '@react-three/drei';
import * as THREE from 'three';
import type { ListingState } from '@/lib/types';

const PIPELINE_STAGES: ListingState[] = [
  'new', 'uploading', 'analyzing', 'awaiting_review', 'in_review', 'approved', 'delivered',
];

const STAGE_LABELS: Record<string, string> = {
  new: 'New', uploading: 'Upload', analyzing: 'Analyze',
  awaiting_review: 'Review', in_review: 'Reviewing', approved: 'Approved', delivered: 'Delivered',
};

interface Props {
  currentState: ListingState;
}

export function PipelineVisualizer({ currentState }: Props) {
  const currentIndex = PIPELINE_STAGES.indexOf(currentState);

  const nodes = useMemo(() => {
    return PIPELINE_STAGES.map((stage, i) => {
      const x = (i - 3) * 1.2;
      const isComplete = i < currentIndex;
      const isCurrent = i === currentIndex;
      const color = isComplete ? '#22C55E' : isCurrent ? '#2563EB' : '#CBD5E1';
      const emissive = isCurrent ? 0.4 : isComplete ? 0.15 : 0;
      const scale = isCurrent ? 0.25 : 0.18;
      return { stage, x, color, emissive, scale, isComplete, isCurrent };
    });
  }, [currentIndex]);

  return (
    <group position={[0, 0, 0]}>
      {nodes.map((node, i) => (
        <group key={node.stage}>
          {/* Node sphere */}
          <mesh position={[node.x, 0, 0]}>
            <sphereGeometry args={[node.scale, 16, 16]} />
            <meshStandardMaterial
              color={node.color}
              emissive={node.color}
              emissiveIntensity={node.emissive}
              metalness={0.4}
              roughness={0.3}
            />
          </mesh>

          {/* Label */}
          <Text position={[node.x, -0.5, 0]} fontSize={0.1} color={node.isCurrent ? '#2563EB' : '#94A3B8'} anchorX="center">
            {STAGE_LABELS[node.stage]}
          </Text>

          {/* Connector line to next node */}
          {i < nodes.length - 1 && (
            <mesh position={[(node.x + nodes[i + 1].x) / 2, 0, 0]}>
              <boxGeometry args={[0.8, 0.02, 0.02]} />
              <meshStandardMaterial
                color={node.isComplete ? '#22C55E' : '#E2E8F0'}
                emissive={node.isComplete ? '#22C55E' : '#000000'}
                emissiveIntensity={node.isComplete ? 0.2 : 0}
              />
            </mesh>
          )}

          {/* Current node glow */}
          {node.isCurrent && (
            <pointLight position={[node.x, 0, 0.5]} intensity={0.8} color="#2563EB" distance={2} />
          )}
        </group>
      ))}
    </group>
  );
}
```

```tsx
import { useState, useEffect, useCallback } from 'react';
import { useParams } from 'next/navigation';
import { motion } from 'framer-motion';
import { apiClient } from '@/lib/api-client';
import type { ListingResponse, AssetResponse, PackageSelection, CreateAssetsResponse, ListingState } from '@/lib/types';

import Link from 'next/link';

export default function ListingDetailPage() {
  const { id } = useParams<{ id: string }>();
  const [listing, setListing] = useState<ListingResponse | null>(null);
  const [assets, setAssets] = useState<AssetResponse[]>([]);
  const [selections, setSelections] = useState<PackageSelection[]>([]);
  const [loading, setLoading] = useState(true);
  const [actionLoading, setActionLoading] = useState(false);
  const [error, setError] = useState('');

  const fetchData = useCallback(async () => {
    try {
      const [l, a, p] = await Promise.all([
        apiClient.get<ListingResponse>(`/listings/${id}`),
        apiClient.get<AssetResponse[]>(`/listings/${id}/assets`),
        apiClient.get<PackageSelection[]>(`/listings/${id}/package`),
      ]);
      setListing(l);
      setAssets(a);
      setSelections(p);
    } catch (err: any) {
      setError(err.message || 'Failed to load listing');
    } finally {
      setLoading(false);
    }

  const handleAction = async (action: 'review' | 'approve') => {
    setActionLoading(true);
    setError('');
    try {
      await apiClient.post(`/listings/${id}/${action}`);
      await fetchData();
    } catch (err: any) {
      setError(err.message || `Failed to ${action}`);

      setActionLoading(false);

      </main>
    </ProtectedRoute>
  );

  const addr = listing.address;
  const meta = listing.metadata as Record<string, unknown>;
  const addressLine = [addr.street, addr.city, addr.state].filter(Boolean).join(', ');

  return (
    <ProtectedRoute>
      <Nav />
      <main className="max-w-5xl mx-auto px-4 sm:px-6 lg:px-8 py-8">
        <Link href="/listings" className="text-sm text-[var(--color-primary)] hover:underline mb-4 inline-block">&larr; Back to listings</Link>

            <h1 className="text-3xl font-bold" style={{ fontFamily: 'var(--font-heading)' }}>{addressLine || 'Untitled listing'}</h1>

              {[meta.beds && `${meta.beds} bed`, meta.baths && `${meta.baths} bath`, meta.sqft && `${meta.sqft} sqft`, meta.price && `$${Number(meta.price).toLocaleString()}`].filter(Boolean).join(' / ')}
            </p>
          </div>
          <Badge label={listing.state} />
        </motion.div>

        {/* 3D Pipeline Status */}
        <GlassCard className="mb-8 p-4">

              <h2 className="text-lg font-semibold mb-4" style={{ fontFamily: 'var(--font-heading)' }}>Assets ({assets.length})</h2>
              {assets.length === 0 ? (
                <p className="text-[var(--color-text-muted)] text-sm mb-4">No assets registered yet.</p>
```
